#include <iostream>
using namespace std;
int max(int x,int y)
{
	if(x>y)
		return x;
	else 
		return y;
}
int main()
{
	int A[50],n,max_value,i;
	cout << "Enter number of elements : ";
	cin >> n;

	for(i=0;i<n;i++)
	{
		cout << "Enter the element " << i+1 << " : ";
		cin >> A[i];
	}
	max_value=A[0];

	for(i=1;i<n;i++)
		max_value=max(max_value,A[i]);
		cout << "\nThe array is : ";

		for(i=1;i<n;i++)
			cout << A[i] << "  ";
			cout << "\nThe largest number is: " << max_value << "\n\n";

	return 0;
}
